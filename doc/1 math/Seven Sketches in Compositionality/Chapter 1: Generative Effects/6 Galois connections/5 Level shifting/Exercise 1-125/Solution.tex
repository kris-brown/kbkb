          \begin{itemize}

            \item and let $Q'=S\times S$
          \end{itemize}
    \item \,$Cl(Q) = \{11,12,22,33\}$\, $\sqsubset$ \,$\leq = \{11,22,33,12,23,13\}$\,
    \item \,$Cl(Q') = Q' = S \times S$\, $\not \sqsubset$ \,$\leq$\,  (reason: \,$(3,1) \in S \times S$ but $(3,1) \not \in \leq$\,)
  \end{enumerate}
